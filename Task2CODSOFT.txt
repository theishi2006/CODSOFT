#include <iostream>
using namespace std;
int main() {
char b[3][3];      // board
char p = 'X';      // current player
int m;             // move
char again;        // play again choice
do {
cout << "Let's have a look on the Board positions :\n";
cout << "1 : 2 : 3\n";
cout << "---------\n";
cout << "4 : 5 : 6\n";
cout << "---------\n";
cout << "7 : 8 : 9\n\n";
cout << "Now, we will begin with the TIC-TAC-TOE game.\n";
for (int i = 0; i < 3; i++) {
for (int j = 0; j < 3; j++) {
b[i][j] = '.';
}
}
bool over = false;
while (!over) {
for (int i = 0; i < 3; i++) {
for (int j = 0; j < 3; j++) {
cout << b[i][j];
if (j < 2) cout << " | ";
}
cout << "\n";
if (i < 2) cout << "---------\n";
}
cout << "Player " << p << " enter (1-9): ";
cin >> m;
if (m < 1 || m > 9) {
cout << "Invalid input!\n";
continue;
}
int r = (m - 1) / 3;
int c = (m - 1) % 3;
if (b[r][c] == 'X' || b[r][c] == 'O') {
cout << "Already filled!\n";
continue;
}
b[r][c] = p;
int win = 0;
for (int i = 0; i < 3; i++) {
if (b[i][0] == p && b[i][1] == p && b[i][2] == p) win = 1;
if (b[0][i] == p && b[1][i] == p && b[2][i] == p) win = 1;
}
if (b[0][0] == p && b[1][1] == p && b[2][2] == p) win = 1;
if (b[0][2] == p && b[1][1] == p && b[2][0] == p) win = 1;
if (win == 1) {
for (int i = 0; i < 3; i++) {
for (int j = 0; j < 3; j++) {
cout << b[i][j];
if (j < 2) cout << " | ";
}
cout << "\n";
if (i < 2) cout << "---------\n";
}
cout << "congrats,Player " << p << " wins!\n";
over = true;
} else {
int draw = 1;
for (int i = 0; i < 3; i++) {
for (int j = 0; j < 3; j++) {
if (b[i][j] == '.') draw = 0;
}
}
if (draw == 1) {
for (int i = 0; i < 3; i++) {
for (int j = 0; j < 3; j++) {
cout << b[i][j];
if (j < 2) cout << " | ";
}
cout << "\n";
if (i < 2) cout << "---------\n";
}
cout << "It's a draw!\n";
over = true;
}
}
if (!over) {
if (p == 'X') p = 'O';
else p = 'X';
}
}
cout << "want to Play again? (y/n): ";
cin >> again;
} while (again == 'y' || again == 'Y');
return 0;
}
